Medical Coding Review Agent YAML
model: medical-coding-review-agent
label:
  en_US: Medical Coding Review Agent
model_type: llm
features:
- agent-thought
- tool-use
description:
  en_US: An AI agent that reviews medical procedure and diagnostic codes for errors, compliance issues, and optimization opportunities.
knowledge_bases:
- icd10
- cpt
- medical-billing
- healthcare-compliance
model_properties:
  mode: chat
  context_size: 16000
parameter_rules:
- name: temperature
  use_template: temperature
  default: 0.7
- name: top_p
  use_template: top_p
  default: 0.9
- name: max_tokens
  use_template: max_tokens
  default: 1024
capabilities:
  code_validation:
    enabled: true
    description: Validates ICD-10 and CPT codes for accuracy and completeness
  code_pairing:
    enabled: true
    description: Checks for proper pairing between diagnostic and procedure codes
  compliance_check:
    enabled: true
    description: Identifies potential compliance issues with coding guidelines
  reimbursement_optimization:
    enabled: true
    description: Suggests coding optimizations to improve reimbursement outcomes
  documentation_guidance:
    enabled: true
    description: Provides guidance on documentation requirements for specific codes
tools:
  icd10_database:
    enabled: true
    description: Tool to query ICD-10 diagnostic code database
  cpt_database:
    enabled: true
    description: Tool to query CPT procedure code database
  coding_guidelines:
    enabled: true
    description: Tool to access coding guidelines and regulations
pricing:
  input: '0.00'
  output: '0.00'
  unit: '0.000001'
  currency: USD
